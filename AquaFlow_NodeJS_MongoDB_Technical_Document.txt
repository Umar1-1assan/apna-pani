Invoice Generation (PDF)               Invoices are generated server-side using PDFKit (a Node.js PDF library) and uploaded to Cloudinary. The PDF URL is stored in the Invoice document and shared with customers via WhatsApp.                       9.1  PDFKit Service                                                // src/services/invoice.service.ts               import PDFDocument from &apos;pdfkit&apos;;               import { uploadPdfBuffer } from &apos;./cloudinary.service&apos;;               import { Invoice } from &apos;../models/Invoice&apos;;               import { Customer } from &apos;../models/Customer&apos;;               import { Supplier } from &apos;../models/Supplier&apos;;                               export async function generateInvoicePdf(invoiceId: string): Promise<string> {                 const invoice  = await Invoice.findById(invoiceId);                 const customer = await Customer.findById(invoice!.customerId);                 const supplier = await Supplier.findById(invoice!.supplierId);                                 const doc = new PDFDocument({ size: &apos;A4&apos;, margin: 50 });                 const buffers: Buffer[] = [];                 doc.on(&apos;data&apos;, (chunk) => buffers.push(chunk));                                 // Header                 doc.fontSize(24).fillColor(&apos;#1E40AF&apos;)                    .text(supplier!.businessName, 50, 50);                 doc.fontSize(12).fillColor(&apos;#475569&apos;)                    .text(`Invoice: ${invoice!.month}/${invoice!.year}`, 50, 85);                                 // Customer Details                 doc.moveDown(2);                 const rows = [                   [&apos;Customer Name&apos;, customer!.fullName],                   [&apos;Phone&apos;,         customer!.phone],                   [&apos;Address&apos;,       customer!.address],                   [&apos;Total Bottles&apos;, String(invoice!.totalBottles)],                   [&apos;Rate/Bottle&apos;,   `PKR ${invoice!.bottlePrice}`],                   [&apos;Previous Dues&apos;, `PKR ${invoice!.previousDues}`],                   [&apos;Discounts&apos;,     `PKR ${invoice!.discounts}`],                   [&apos;Total Amount&apos;,  `PKR ${invoice!.totalBottles * invoice!.bottlePrice + invoice!.previousDues - invoice!.discounts}`],                 ];                                 rows.forEach(([label, value]) => {                   doc.fontSize(11).fillColor(&apos;#475569&apos;).text(label, 50, doc.y, { width: 200 })                      .fontSize(11).fillColor(&apos;#1E293B&apos;).text(value, 260, doc.y - 14);                   doc.moveDown(0.5);                 });                                 doc.end();                                 await new Promise(r => doc.on(&apos;end&apos;, r));                 const buffer = Buffer.concat(buffers);                                 // Upload to Cloudinary                 const result: any = await uploadPdfBuffer(                   buffer,                   `${supplier!._id}/invoices`,                   `invoice_${customer!._id}_${invoice!.year}_${invoice!.month}`                 );                                 // Save URL back to Invoice                 await Invoice.findByIdAndUpdate(invoiceId, { pdfUrl: result.secure_url });                 return result.secure_url;               }                             10.
